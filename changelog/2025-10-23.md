## 主要修改項目

### 1. 添加 asyncio.Lock 防止 Race Condition

```python
# 修改前
def __init__(self):
    self.active_connections: Dict[int, List[WebSocket]] = {}

# 修改後
def __init__(self):
    self.active_connections: Dict[int, List[WebSocket]] = {}
    self._lock = asyncio.Lock()  # 新增:防止多個用戶同時操作導致衝突
```


---

### 2. 防止重複連接

```python
# 修改前
async def connect(self, group_id: int, websocket: WebSocket):
    await websocket.accept()
    if group_id not in self.active_connections:
        self.active_connections[group_id] = []
    self.active_connections[group_id].append(websocket)  # 可能重複添加

# 修改後
async def connect(self, group_id: int, websocket: WebSocket):
    await websocket.accept()
    async with self._lock:  # 使用鎖保護
        if group_id not in self.active_connections:
            self.active_connections[group_id] = []
        
        # 檢查是否已存在
        if websocket not in self.active_connections[group_id]:
            self.active_connections[group_id].append(websocket)
            logger.info(f"User connected to group {group_id}. Total: {len(self.active_connections[group_id])}")
```

### 4. 改進 disconnect 方法

```python
# 修改前
async def disconnect(self, group_id: int, websocket: WebSocket):
    if group_id in self.active_connections:
        if websocket in self.active_connections[group_id]:
            self.active_connections[group_id].remove(websocket)
        if not self.active_connections[group_id]:
            del self.active_connections[group_id]

# 修改後
async def disconnect(self, group_id: int, websocket: WebSocket):
    async with self._lock:  # 使用鎖保護
        if group_id in self.active_connections:
            if websocket in self.active_connections[group_id]:
                self.active_connections[group_id].remove(websocket)
                logger.info(f"User disconnected from group {group_id}. Remaining: {len(self.active_connections[group_id])}")
            
            if not self.active_connections[group_id]:
                del self.active_connections[group_id]
```
